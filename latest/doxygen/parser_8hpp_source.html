67 //-----------------------------------------------------
68 // Helper: Split a qualified name "prefix:local" into namespace and local name.
69 // If no colon exists, returns { name, "" }.
70 // The returned pair is in the form (local name, namespace)
71 // because builder calls are of the form begin(localName, ns).
72 static std::pair<std::string_view, std::string_view> split_namespace(std::string_view qualified) {
73 size_t pos = qualified.find(':');
74 if (pos != std::string_view::npos) {
75 // Namespace is the left side, local name is right side.
76 return { qualified.substr(pos + 1), qualified.substr(0, pos) };
77 }
78 return { qualified, "" };
79 }

118 // Main element parser. It assumes that a '<' has already been consumed.
119 template<bool ROOT=false>
120 void parseElement() {
121 skip_whitespace();
122 if (pos_ >= data_.size())
123 return;
124
125 // Check for processing instruction <? ... ?>
126 if (peek('?')) {
127 ++pos_; // skip '?'
128 // Read processing instruction until "?>"
129 size_t procEnd = data_.find("?>", pos_);
130 if (procEnd == std::string_view::npos)
131 throw std::runtime_error("Unterminated processing instruction.");
132 auto procContent = data_.substr(pos_, procEnd - pos_);
133 if constexpr (Builder_t::configs.raw_strings ) builder_.proc(procContent);
134 else builder_.proc(serialize::inplace_unescape_xml(procContent));
135 pos_ = procEnd + 2;
136 return;
137 }
138
139 // Check for comment or CDATA (both start with "!")
140 if (peek('!')) {
141 ++pos_; // skip '!'
142 // Comment: <!-- ... -->
143 if (data_.substr(pos_, 2) == "--") {
144 pos_ += 2;
145 size_t commentEnd = data_.find("-->", pos_);
146 if (commentEnd == std::string_view::npos)
147 throw std::runtime_error("Unterminated XML comment.");
148 auto commentContent = data_.substr(pos_, commentEnd - pos_);
149 if constexpr (Builder_t::configs.raw_strings ) builder_.comment(commentContent);
150 else builder_.comment(serialize::inplace_unescape_xml(commentContent));
151 pos_ = commentEnd + 3;
152 return;
153 }
154 // CDATA: <![CDATA[ ... ]]>
155 else if (data_.substr(pos_, 7) == "[CDATA[" && !ROOT) {
156 pos_ += 7; // skip "[CDATA["
157 size_t cdataEnd = data_.find("]]>", pos_);
158 if (cdataEnd == std::string_view::npos)
159 throw std::runtime_error("Unterminated CDATA section.");
160 auto cdataContent = data_.substr(pos_, cdataEnd - pos_);
161 builder_.cdata(cdataContent); // CDATA content provided as-is.
162 pos_ = cdataEnd + 3;
163 return;
164 } else if(!ROOT){
165 // Other declarations - skip until '>'
166 get_until('>');
167 consume('>');
168 return;
169 } else{
170 throw std::runtime_error("Node type not allowed in the document root.");
171 }
172 }
173
174 // Standard element:
175 // Parse element qualified name. Allowed characters: alphanumeric, '_', ':', '-'
176 auto qualifiedName = get_while([](char c) {
177 return std::isalnum(static_cast<unsigned char>(c)) || c == '_' || c == ':' || c == '-';
178 });
179 auto [localName, ns] = split_namespace(qualifiedName);
180
181 builder_.begin(localName, ns);
182
183 // Parse attributes (if any)
184 while (true) {
185 skip_whitespace();
186 // End of tag? Either '>' or '/>'
187 if (data_.substr(pos_, 2) == "/>" || peek('>'))
188 break;
189
190 // Parse attribute qualified name.
191 auto attrQualified = get_while([](char c) {
192 return std::isalnum(static_cast<unsigned char>(c)) || c == '_' || c == ':' || c == '-';
193 });
194 auto [attrLocal, attrNs] = split_namespace(attrQualified);
195
196 skip_whitespace();
197 if (!consume('='))
198 throw std::runtime_error("Expected '=' after attribute name.");
199 skip_whitespace();
200 // Expect a quoted attribute value.
201 char quote = 0;
202 if (peek('\"') || peek('\'')) {
203 quote = data_[pos_];
204 ++pos_;
205 } else {
206 throw std::runtime_error("Expected quote for attribute value.");
207 }
208
209 size_t attrValueStart = pos_;
210 while (pos_ < data_.size() && data_[pos_] != quote)
211 ++pos_;
212 if (pos_ >= data_.size())
213 throw std::runtime_error("Unterminated attribute value.");
214 auto attrValue = data_.substr(attrValueStart, pos_ - attrValueStart);
215 if (!consume(quote))
216 throw std::runtime_error("Expected closing quote for attribute value.");
217
218 // Call builder's attr with local name and corresponding namespace.
219 if constexpr (Builder_t::configs.raw_strings ) builder_.attr(attrLocal, attrValue, attrNs);
220 else builder_.attr(attrLocal, serialize::inplace_unescape_xml(attrValue), attrNs);
221
222 }
223
224 // Self-closing element?
225 if (data_.substr(pos_, 2) == "/>") {
226 pos_ += 2;
227 builder_.end();
228 return;
229 }
230
231 // Otherwise, consume the '>' and open the element.
232 if (!consume('>'))
233 throw std::runtime_error("Expected '>' after tag name and attributes.");
234
235 // Parse content within the element until encountering a matching end tag.
236 while (pos_ < data_.size()) {
237 skip_whitespace();
238 if (pos_ >= data_.size())
239 throw std::runtime_error("Unexpected end of XML content.");
240
241 if (peek('<')) {
242 // Check for closing tag.
243 if (data_.substr(pos_, 2) == "</") {
244 pos_ += 2; // skip "</"
245 // Skip the qualified name inside end tag.
246 get_until('>');
247 if (!consume('>'))
248 throw std::runtime_error("Expected '>' in closing tag.");
249 builder_.end();
250 break;
251 } else {
252 // Child element or special node.
253 ++pos_; // skip '<'
254 parseElement(); // recursive call
255 }
256 } else {
257 // Process text content until next '<'
258 size_t textStart = pos_;
259 while (pos_ < data_.size() && data_[pos_] != '<')
260 ++pos_;

262 std::string_view unescapedText;
263
264 if constexpr (Builder_t::configs.raw_strings )unescapedText=textContent;
265 else unescapedText = serialize::inplace_unescape_xml(textContent);
266
267 if (!unescapedText.empty() &&
268 unescapedText.find_first_not_of(" \t\r\n") != std::string::npos)
269 {
