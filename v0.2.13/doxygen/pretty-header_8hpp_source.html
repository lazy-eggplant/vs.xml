32// Return an ANSI color code based on the byte offset.
33inline const char* getColorForOffset(size_t offset) {
34 if (offset < 4) return colors::red; // magic[4]
35 else if (offset == 4) return colors::green; // format_major
36 else if (offset == 5) return colors::yellow; // format_minor
37 else if (offset == 6) return colors::blue; // configs (1 bytes)
38 else if (offset == 7) return colors::magenta; // endianess/res0 (1 byte)
39 else if (offset >= 8 && offset < 12) return colors::cyan; // bitfields for sizes (4 bytes)
40 else if (offset >= 12 && offset < 14) return colors::white; // docs_count (2 bytes)
41 else if (offset >= 14 && offset < 16) return colors::brightYellow; // res[2]
42 else if (offset >= 16 && offset < 16 +
43 sizeof(VS_XML_NS::xml_count_t)) return colors::brightMagenta; // length_of_symbols (8 bytes)
44 else return colors::reset; // sections & others
45}

47template<uint N>
48void print_header_colors(std::span<const uint8_t> data) {
49 constexpr size_t groupSize = N; // We'll print 32 bytes per line
50
51 for (size_t offset = 0; offset < data.size(); offset += groupSize) {
52 // Print the offset (address) header
53 std::print("{:04x}: ", offset);
54
55 // Determine how many bytes are in this group.
56 size_t bytesInGroup = std::min(groupSize, data.size() - offset);
57
58 // Print hexadecimal bytes with color coding.
59 for (size_t i = 0; i < groupSize; ++i) {
60 if (i < bytesInGroup) {
61 size_t byte_offset = offset + i;
62 const char* color = getColorForOffset(byte_offset);
63 // Print the colored two-digit hex and then reset color after each byte.
64 std::print("{}{:02x}{} ", color, data[byte_offset], colors::reset);
65 } else {
66 // Add spacing if no data.
67 std::print(" ");
68 }
69 }
70
71 // Print ASCII representation
72 std::print(" | ");
73 for (size_t i = 0; i < bytesInGroup; ++i) {
74 uint8_t byte = data[offset + i];
75 char ch = std::isprint(byte) ? static_cast<char>(byte) : '.';
76 // Use same color as the hex part.
77 const char* color = getColorForOffset(offset + i);
78 std::print("{}{}{}", color, ch, colors::reset);
79 }
80 std::print("\n");
81 }
