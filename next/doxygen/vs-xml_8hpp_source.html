90 constexpr std::optional<std::string_view> validate_xml_label(std::string_view str){
91 int pos = 0;
92 for(auto& c : str){
93 if(pos==0 && (c=='_' or (c>'a' && c<'z') or (c>'A' && c<'Z'))){/*OK*/}
94 //In theory some intervals of utf8 should be negated. But this filter is good enough for now.
95 else if(pos!=0 && (c=='_' or c=='.' or c=='-' or (c>'0' && c<'9') or (c>'a' && c<'z') or (c>'A' && c<'Z') or (c>127))){/*OK*/}
96 else{
97 //std::print("{} @ pos {}----- {} {} {} \n",(int)c, pos, str, (void*)str.data(), str.length());
98 return {};
99 }
100 pos++;
101 }
102 return str;
103 }

149 constexpr ret_t to_xml_text(std::string_view str){
150 int rule_a = 0; //Offset for 1-character escape sequiences
151 int rule_b = 0; //Intermediate steps of ]]>
152 int rule_c = 0; //Offset for rule_b
153 for(auto& c : str){
154 if(c=='<')rule_a+=sizeof("&lt;")-1-1;
155 else if(c=='&')rule_a+=sizeof("&amp;")-1-1;
156 else if(rule_b==0 && c==']')rule_b=1;
157 else if(rule_b==1 && c==']')rule_b=2;
158 else if(rule_b==2 && c=='>'){rule_b=0;rule_c+=sizeof("&gt;")-1-1;} //]]> not allowed in text, it must be escaped as `]]&gt;`.
159 else rule_b=0;
160 //Maybe some config to enable the others as well even if not needed for a correct XML serialization?
161 }
162 if(rule_a==0 && rule_c==0)return str;
163 else{
164 std::string tmp;
165 tmp.reserve(str.length()+rule_a+rule_c);
166 int rule_b = 0;
167 for(auto& c : str){
168 if(rule_b==0 && c==']')rule_b=1;
169 else if(rule_b==1 && c==']')rule_b=2;
170 else if(rule_b==2 && c=='>'){rule_b=0;tmp+="]]&gt;";} //]]> not allowed in text, it must be escaped as `]]&gt;`.
171 else if(rule_b==1){rule_b=0;tmp+="]";}
172 else if(rule_b==2){rule_b=0;tmp+="]]";}
173 else if(c=='<')tmp+="&lt;";
174 else if(c=='&')tmp+="&amp;";
175 else tmp+=c;
176 }
177 return tmp;
178 }

755 private:
756 std::vector<uint8_t> buffer;
757 bool open = true; //True if the tree is still open to append things.
758 bool attribute_block = false; //True after a begin to add attributes. It is automatically closed when any other command is triggered.
759
760 std::stack<std::pair<ptrdiff_t,ptrdiff_t>> stack;
761
762 template<typename T>
763 error_t leaf(std::string_view value){
764 if(open==false)return error_t::TREE_CLOSED;
765 attribute_block=false;
766
767 buffer.resize(buffer.size()+sizeof(T));
768
769 auto& old_ctx = stack.top();
770
771 unknown_t* prev = old_ctx.second!=-1?(unknown_t*)(buffer.data()+old_ctx.second):nullptr;
772
773 T* tmp_node = new ((node_t*) & (uint8_t&) *( buffer.end()-sizeof(T) )) T(label_offset,value);
774
775 if(prev!=nullptr){
776 prev->set_next((unknown_t*)tmp_node);
777 tmp_node->set_prev(prev);
778 }
779 old_ctx.second = (uint8_t*)tmp_node-buffer.data();
780
781 return error_t::OK;
782 }
783
784 protected:
785 void* label_offset = nullptr;
786
787 std::expected<Tree,error_t> close(std::vector<uint8_t>&& symbols){
788 if(open==false)return std::unexpected(error_t::TREE_CLOSED);
789 open=false;
790 if(stack.size()!=1)return std::unexpected(error_t::MISFORMED);
791 stack.pop();
792
793 buffer.resize(buffer.size()+sizeof(eot_t));
794 new ((node_t*) & (uint8_t&) *( buffer.end()-sizeof(eot_t) )) eot_t();
795
796 return Tree(std::move(buffer),std::move(symbols));
797 }

799 public:

801 Builder(){stack.push({0,-1});}

803 error_t begin(std::string_view name, std::string_view ns=""){
804 if(open==false)return error_t::TREE_CLOSED;
805
806 buffer.resize(buffer.size()+sizeof(node_t));
807
808 auto& old_ctx = stack.top();
809
810 node_t* parent = (node_t*)(buffer.data()+old_ctx.first);
811 unknown_t* prev = old_ctx.second!=-1?(unknown_t*)(buffer.data()+old_ctx.second):nullptr;
812
813 //Emplace node
814 node_t* tmp_node = new ((node_t*) & (uint8_t&) *( buffer.end()-sizeof(node_t) )) node_t(label_offset,parent,ns,name);
815
816 if(prev!=nullptr){
817 prev->set_next((unknown_t*)tmp_node);
818 tmp_node->set_prev(prev);
819 }
820 old_ctx.second = (uint8_t*)tmp_node-buffer.data();
821
822 stack.push({((uint8_t*)tmp_node-(uint8_t*)buffer.data()),-1});
823 attribute_block=true;
824
825 return error_t::OK;
826 }
827
828 error_t end(){
829 if(open==false)return error_t::TREE_CLOSED;
830 if(stack.size()<=1)return error_t::STACK_EMPTY;
831
832 attribute_block=false;
833
834 auto& ctx = stack.top();
835 node_t* parent = (node_t*)(buffer.data()+ctx.first);
836 parent->_size=buffer.size()-ctx.first;
837
838 stack.pop();
839
840 return error_t::OK;
841 }

847 buffer.resize(buffer.size()+sizeof(attr_t));

852
853 //Emplace node

855
856 parent->attrs_count++;
857
858 return error_t::OK;
